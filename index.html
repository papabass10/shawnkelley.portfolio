Karaoke page queries

These are the querying and sorting functions for the Pro Show DJs Karaoke page. The functions were written in PHP and recieve the initial parameters from an AJAX call to the form data on the page.

First the search creation function:

									
function karaoke_search($options){
	global $wpdb;
	$columns = $options['columns'];
	$searchValue = $options['searchValue'];
	$duetValue = $options['duetValue'];
	$vocalValue = $options['vocalValue'];
	$pageSize = $options['pageSize'];
	$pageNumber = $options['pageNumber'];
	$sortResultsBy = $options['sortResultsBy'];
	$ascDesc = $options['ascDesc'];
		$columnWhere = parse_where($columns, $searchValue);
	$duetVocals = parse_where_and($duetValue, $vocalValue);
	$offsetNumber = $pageSize * ($pageNumber-1);
	$query = $wpdb->get_results(
		"SELECT *
		FROM karaokesongs
		WHERE ({$columnWhere}) {$duetVocals}
		ORDER BY {$sortResultsBy} {$ascDesc}
		LIMIT {$pageSize}
		OFFSET {$offsetNumber}
		",
		ARRAY_A
	);
	return $query;
}

Then the code for parsing the WHERE and ORDER BY strings to allow for searcing by artist or title, and sorting by duets, vocals, both, or neither

									
function parse_where($columns, $searchValue){
	$columnWhere = "";
	foreach($columns as $column){
		$columnWhere .= $column . " LIKE '%" . $searchValue . "%'";
		if($column !== end($columns)) {
			$columnWhere .= " OR ";
		}
	}
	return $columnWhere;
}

function parse_where_and($duetValue, $vocalValue){
	$duetVocals = "";
	if($duetValue == 'duet'){
		$duetVocals .= " AND duet <> ''";
	}
	elseif($duetValue == 'no-duets'){
		$duetVocals .= " AND duet = ''";
	}

	if($vocalValue == 'vocal'){
		$duetVocals .= " AND vocal <> ''";
	}
	elseif($vocalValue == 'no-vocals'){
		$duetVocals .= " AND vocal = ''";
	}
	return $duetVocals;
}
